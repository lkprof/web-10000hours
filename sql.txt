create table users(
	user_id int auto_increment,
	name varchar(20),
	email varchar(30),
	password varchar(20),
	sex varchar(4),
	primary key(id)
);

create table subjects(
	subject_id int auto_increment,
	title varchar(20),
	context varchar(20),
	posted_on datetime,
	primary key(id),
	user_id int,
	foreign key(user_id) references users(user_id)
);
create table spend_on_subject(
	spend_id int auto_increment,
	spend_time int,
	summary text,
	posted_on datetime,
	primary key(spent_id),
	foreign key(subject_id) references subjects(subject_id)
);
